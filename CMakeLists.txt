project(mpu9255)

  roscpp
  rospy
  sensor_msgs
  tf
)

  ${catkin_INCLUDE_DIRS}
)

add_executable(mpu9255_node src/mpu9255_node.cpp)
target_link_libraries(mpu9255_node ${catkin_LIBRARIES})
target_link_libraries (mpu9255_node wiringPi)
